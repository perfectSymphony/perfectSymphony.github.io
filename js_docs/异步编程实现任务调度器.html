    <script>
      let box = document.getElementById('box')
      box.className = 'red';
      function task(t){
        return new Promise(function(resolve, reject){
          setTimeout(resolve, t)
        })
      }
      function stat(){
        task(3000).then(() => {
          box.className = 'yellow'
          return task(1000)
        }).then(() => {
          box.className = 'green'
          return task(2000)
        }).then(() => {
          box.className = 'red'
          stat()
        })
      }
      stat()
